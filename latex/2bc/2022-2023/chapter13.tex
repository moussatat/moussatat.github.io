\section{Produit d'un vecteur par un rĂ©el}

\begin{definition}[interprĂ©tation gĂ©omĂ©trique] $\vv{u}\neq \vv{0}$. 
Le produit d'un rĂ©el $k$ par un vecteur $\vv{u}$ est un vecteur notĂ© $k\vv{u}$ : 
\begin{description}
\item[P1]  $0\vv{u}=\vv{0}$.
\item[P2]  Si $k \neq  0$,  \og $k\vv{u}$ \fg{} dĂ©signe le vecteur :
\begin{enumerate}[label=\textbullet, leftmargin=*]
\item ayant mĂªme direction que $\vv{u}$ 
\item $\norm{k\vv{u}}=\abs{k}\times \norm{\vv{u}}$
\item Si $k>0$ alors $k\vv{u}$ et $\vv{u}$ ont \textbf{mĂªme sens}.\\
Si $k<0$ alors $k\vv{u}$ et $\vv{u}$ sont de \textbf{sens contraires}
\end{enumerate}
\end{description} 
\end{definition}


\vfill

\begin{definition}[vecteurs opposĂ©s]  $\vv{v}=-\vv{u}$ signifie que les vecteurs $\vv{u}$ et $\vv{v}$  ont  mĂªme direction ,  mĂªme norme mais sont de sens contraires. 
En particulier : 	$-\vv{AB}=\vv{BA}$
\end{definition}

\begin{enumerate}[label=\alph*), leftmargin=*]
\item Ă‰numĂ©rer 2 Ă©galitĂ©s de vecteurs.
\vfill
\item Ă‰numĂ©rer 3 paires de vecteurs opposĂ©s $\vv{v}=-\vv{u}$
\vfill
\item Ă‰numĂ©rer 3 paires de vecteurs qui peuvent s'Ă©crire $\vv{v}=2\vv{u}$
\vfill
\item Ă‰numĂ©rer 3 paires de vecteurs qui peuvent s'Ă©crire $\vv{v}=\frac{1}{2}\vv{u}$
\vfill
\end{enumerate}
\end{example}

\begin{definition} Soit trois points non alignĂ©s $O$, $I$ et $J$. Les vecteurs $\vv{OI}=\vv{\imath}$ et $\vv{OJ}=\vv{\jmath}$ ne sont pas colinĂ©aires et forment une \textbf{base}.\\

\begin{enumerate}[leftmargin=*,label=\textbullet]
	\item Tout point $P$ admet un unique couple de coordonnĂ©es $(x_P~;~y_P)$   :
	\setlength{\abovedisplayskip}{3pt}
	\setlength{\belowdisplayskip}{3pt}
	$$\vv{OP}=x_P\vv{\imath}+y_P\vv{\jmath}$$  
	\item \noindent Tout vecteur $\vv{u}$ admet un unique couple de coordonnĂ©es $\begin{pmatrix}
		x\\ y
	\end{pmatrix}$ dans la \textbf{base} $(\vv{\imath}~;~\vv{\jmath})$ tel que 
	$\vv{u}=x\vv{\imath}+y\vv{\jmath}$ 
\end{enumerate}  
\end{definition}
